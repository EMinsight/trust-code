#!/bin/bash fichier=$TRUST_TMP/Maillages_VDF.html # Ecriture du fichier : echo $ECHO_OPTS " TRUST TRUST 


Fichiers GIBI pour creer des maillages structures VDF 


" > $fichier
cd $TRUST_ROOT/bin
for fic in ../GIBI/Tests_VDF/*.gibi
do
   REP=`dirname $fic`
   NOM=`basename $fic`
   NOM=${NOM%.gibi}
#   com=`head -1 $fic`
   if [ -f $REP/$NOM.gif ]
   then
      echo $ECHO_OPTS "$NOM\c" >> $fichier
   else
      echo $ECHO_OPTS "$NOM\c" >> $fichier
   fi
#   echo $ECHO_OPTS "$com" >> $fichier
   echo $ECHO_OPTS "
" >> $fichier
done
echo $ECHO_OPTS "
 " >> $fichier
new=$TRUST_ROOT/IHM/Maillages_VDF.html
if [ "`diff $fichier $new`" != "" ]
then
   CHECKOUT $new
   cat $fichier>$new
   CHECKIN $new
fi
