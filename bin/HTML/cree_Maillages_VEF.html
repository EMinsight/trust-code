#!/bin/bash fichier=$TRUST_TMP/Maillages_VDF.html # Ecriture du fichier : echo " TRUST TRUST 


Fichiers GIBI de maillages non-structures VEF


" > $fichier
cd $TRUST_ROOT/bin
for fic in ../GIBI/Tests_VEF/*.GIBI
do
   REP=`dirname $fic`
   NOM=`basename $fic`
   NOM=${NOM%.GIBI}
#   com=`head -1 $fic`
   if [ -f $REP/$NOM.gif ]
   then
      echo $ECHO_OPTS "$NOM\c" >> $fichier
   else
      echo $ECHO_OPTS "$NOM\c" >> $fichier
   fi
#   echo "$com" >> $fichier
   echo "
" >> $fichier
done
echo "
 " >> $fichier
new=$TRUST_ROOT/../Doc_TRUST/IHM/Maillages_VEF.html
if [ "`diff $fichier $new`" != "" ]
then
   CHECKOUT $new
   cat $fichier>$new
   CHECKIN $new
fi
