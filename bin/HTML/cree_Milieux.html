#!/bin/bash fichier=$TRUST_ROOT/../Doc_TRUST/IHM/.Milieux.html # Ecriture du fichier : cree_en_tete.ihm > $fichier echo "
Physical caracteristics of some fluids and solids
" >> $fichier
cd $TRUST_ROOT/doc
for fic in ../../Doc_TRUST/Milieux_reference/*.prop
do
   REP=`dirname $fic`
   NOM=`basename $fic`
   NOM=${NOM%.prop}
   com=$NOM
   if [ -f $REP/.$NOM.prop ]
   then
      com=`head -1 $REP/.$NOM.prop`
   fi
   if [ -f $REP/$NOM.gif ]
   then
      echo $ECHO_OPTS "$com" >> $fichier
   else
      echo $ECHO_OPTS "$com" >> $fichier
   fi
   echo $ECHO_OPTS "
" >> $fichier
done
echo $ECHO_OPTS "
 " >> $fichier
milieu=$TRUST_ROOT/../Doc_TRUST/IHM/Milieux.html
if [ "`diff $fichier $milieu`" != "" ]
then
   CHECKOUT $milieu
   cat $fichier>$milieu
   CHECKIN $milieu
fi
rm $fichier
