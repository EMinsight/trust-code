#!/bin/bash ETUDE_TRUST=`QUELLE_ETUDE.ihm` export ETUDE_TRUST NOMETUDE=`echo $ECHO_OPTS $ETUDE_TRUST | $TRUST_Awk '{n=split($0,a,"/");print a[n]}'` cd $ETUDE_TRUST fichier=$ETUDE_TRUST/.Liste.geom.html # Ecriture du fichier : echo $ECHO_OPTS " STUDY: $NOMETUDE TRUST 


List of study $NOMETUDE mesh files 


" > $fichier
for fic in `ls $ETUDE_TRUST/*.geom 2>/dev/null`
do
   REP=`dirname $fic`
   NOM=`basename $fic`
   NOM=${NOM%.geom}
   echo $ECHO_OPTS "
$NOM.geom
\n" >> $fichier
   if [ -s .$NOM.geom.info ]
   then
      echo $ECHO_OPTS "Click here to see the mesh of the file $NOM.geom" >> $fichier
      echo $ECHO_OPTS " " >> $fichier
      cat $REP/.$NOM.geom.info | $TRUST_Awk '(NR>1) { print $0 }' >> $fichier
   else
      echo $ECHO_OPTS "Afficher $NOM.geom" >> $fichier
   fi
   echo $ECHO_OPTS "
" >> $fichier
done
echo $ECHO_OPTS "
 " >> $fichier
